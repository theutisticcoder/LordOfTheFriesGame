const InitGame = async () => { let model, camera; camera = { 0 }; camera.position = { 0, 10, 10 }; // Camera position camera.target = { 0, 0, 0 }; // Camera looking at point camera.up = { 0, 1, 0 }; // Camera up vector (rotation towards target) camera.fovy = 45; // Camera field-of-view Y camera.projection = CAMERA_PERSPECTIVE; InitWindow(800, 450) model = LoadModel("./abandoned_mcdonalds.glb") } const UpdateGame = (ts) => { BeginDrawing() ClearBackground(RED) BeginMode3D(camera); DrawModel(model, {0,0,0}, 1, BLACK) UpdateCamera(&camera, CAMERA_FIRST_PERSON); EndMode3D() DrawFPS(10, 10) EndDrawing() }
